-- ======================================
-- 数据库和表的创建
-- ======================================
create database test
use test
create table test(id int, name string)


-- ======================================
-- 数据插入测试
-- ======================================
insert into test values(1, "aaa")
insert into test values(2, "bbb")
insert into test values(3, "ccc")
insert into test values(4, "aaa")


-- ======================================
-- 查询功能测试
-- ======================================
-- 查询所有数据
select * from test

-- 条件查询：查找id=1的数据
select * from test where id = 1

-- 条件查询：查找name="aaa"的数据
select * from test where name = aaa

-- 条件查询：查找id>1的数据
select * from test where id > 1


-- ======================================
-- 更新功能测试
-- ======================================
-- 更新数据：将id=1的name字段更新为"ddd"
update test set name = "ddd" where id = 1

-- 验证更新是否成功
select * from test where id = 1


-- ======================================
-- 删除功能测试
-- ======================================
-- 删除数据：删除id=1的数据
delete from test where id = 1

-- 验证数据是否已删除
select * from test where id = 1


-- ======================================
-- 清理测试环境
-- ======================================
drop table test
drop database test
